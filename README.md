# zuri_task-2


# step 1
- create a .env file and it should have all this
    APPLICATION_PORT=9004
    DEV_DB=zuritask2
    DEV_USER=root
    DEV_PASSWORD=""
    DEV_HOST=localhost

# step 2
- run npm install

# step 3
- using MySQL, create database and name it "zuritask2"
# step 4
- run "npm start"

# To run test, run "npm run test"

# To view documentation, open http://localhost:9004/swagger
